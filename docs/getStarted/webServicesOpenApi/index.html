Web Services with Open API
Implementation of Web Services using Open API (Swagger JSON).
Steps to create Web Service using Open API in Simplity
1. Get simplity kernel
For Maven users-
<dependency>
<groupId>org.simplity<groupId>
<artifactIdId>simplity</artifactId>
<version>1.2</version>
</dependency>
Plain Java-
<dependency>
<groupId>org.simplity</groupId>
<artifactIdId>simplity</artifactId>
<version>1.2</version>
</dependency>
2. Creating project
  • Create a new "Maven Project" in eclipse
  • Configure web.xml to have the following entry
    <servlet>
      <servlet-name>Startup</servlet-name>
      <servlet-class>org.simplity.rest.Startup</servlet-class>
      <load-on-startup>1</load-on-startup>
    </servlet>
    <servlet>
      <servlet-name>RestSimplity</servlet-name>
      <servlet-class>org.simplity.rest.Serve</servlet-class>
    </servlet>
    <servlet-mapping>
      <servlet-name>RestSimplity</servlet-name>
      <url-pattern>/app/*</url-pattern>
    </servlet-mapping>

3. Application Configuration
  • Create component folder structure under WEB-INF as follows:
    • api
    • comp
      • dt
      • msg
      • rec
      • service
        • tp
      • sql

  • Configure application.xml under WEB-INF/comp as follows.
    <?xml version="1.0" encoding="UTF-8"?>
    <application
            xmlns="http://www.simplity.org/schema"
            applicationId="ContractService"
            autoLoginUserId="100"
            dbVendor="oracle"
            dbDriverClassName="oracle.jdbc.OracleDriver"
            connectionString="database connection string"
            logSqls="true">
    </application>
      
    • This is the root file of simplity which helps to configure various features of Simplity. Here "applicationId" is a mandatory field which represents the application name. applicationId should be a unique name for the application.
    • NOTE: To get more help, click ctrl + tab from the eclipse editor.
  • Create Open API under WEB-INF/api as follows.
    {
        "swagger": "2.0",
        "info": {
            "description": "",
            "version": "2.0",
            "title": "org.simplity.apiscdb"
        },
        "host": "localhost:8070",
        "basePath": "/app/openapi",
        "schemes": [ "http" ],
        "consumes": [ "application/json" ],
        "produces": [ "application/json" ],
        "paths": {
            "/contract": {
                "get": {
                    "tags": [ "contract" ],
                    "operationId": "contractFind",
                    "summary": "contractFind",
                    "description":"",
                    "deprecated": false,
                    "responses": {
                        "200": {
                            "description": "List of contracts",
                            "schema": {
                                "$ref": "#/definitions/ContractHeaders"
                            }
                        }
                    }
                }
            }
        },
        "definitions": {
            "ContractHeaders": {
                "type": "object",
                "x-fieldName": "ContractHeaders",
                "properties": {
                    "contractHeaders": {
                        "type": "array",
                        "items": {
                            "$ref": "#/definitions/ContractHeader"
                        }
                    }
                }
            },
            "ContractHeader": {
                "type": "object",
                "properties": {
                    "bpContractNum": {
                        "type": "string"
                    },
                    "description": {
                        "type": "string"
                    },
                    "assetName": {
                        "type": "string"
                    }
                }
            },
            "ErrorResponse": {
                "x-fieldName": "ErrorResponse",
                "type": "object",

        }
    }

    • host : host name and port number
    • basePath : path to be appended with host
    • operationId : Simplity service name
    • For more details, refer the "https://swagger.io/"
  • Create Simplity record under WEB-INF/comp/rec as follows.
    <?xml version="1.0" encoding="UTF-8" standalone="no"?>
    <record name="ContractHeader"
            recordType="storage"
            tableName="CONTRACTHEADER"
            keyToBeGenerated="false"
            okToSelectAll="true"
            xmlns="http://www.simplity.org/schema">
      <fields>
        <field columnName="BP_CONTRACT_NUM" dataType="_text" fieldType="primaryKey" name="bpContractNum"         sqlTypeName="VARCHAR"/>
        <field columnName="DESCRIPTION" dataType="_text" fieldType="data" name="description" sqlTypeName="VARCHAR"/>
        <field columnName="ASSET_NAME" dataType="_text" fieldType="data" name="assetName" sqlTypeName="VARCHAR"/>
      </fields>
    </record>

    • Simplity records are kept under the comp/rec folder. This is the main part of data model. Data structures that are used as input to, or output from a service are modeled as records. Following are the fields of simplity record:
    • name (mandatory field) - name of the record or entity
    • moduleName (optional field) - sub-folder name inside rec folder. In this example(rec/login/user.xml) record name is user.xml and module name login.
    • recordType (mandatory field) - STORAGE (represents a row of a table being stored in an rdbms), VIEW (a view involving one or more storage records), STRUCTURE (record structure used by the application, but is not linked to any stored data).
    • tableName (optional field) - name of the db table, if recordType is STORAGE or VIEW.
    • keyToBeGenerated (optional field) - true/false (if any table column value to be generated automatically)
    • okToSelectAll (optional field) - true/false(set true to select all rows from the table)
  • Create Simplity service under WEB-INF/comp/service/tp as follows.
    <?xml version="1.0" encoding="UTF-8" standalone="no"?>
    <service xmlns="http://www.simplity.org/schema"
            name="contractFind"
            dbAccessType="readOnly">
        <actions>
            <filter filterRecordName="filteredContractHeaders" outputSheetName="contractHeaders">
            </filter>
            <log names="contractHeaders"/>
        </actions>
        <outputData>
            <outputRecords>
                <outputRecord sheetName="contractHeaders" />
            </outputRecords>
        </outputData>
    </service>

3. Invoke Simplity Web Service
  • Enter the below URL from the web browser,
    "http://localhost:8070/ContractService/app/openapi/contract/"

    • Note :
      • ContractService - application name.
      • /app/openapi - basePath.
      • /contract - path to the particular service.
4. Create client to invoke web service
  • Here, angular js code used to call web service.
  • Create Simplity client as follows.
    openapi = angular.module('openapi', []);
    openapi.controller('OpenApiCtrl', function($scope, $http) {
        $scope.getContracts = function() {
            $scope.contracts = [];
            var req = {
                method: 'GET',
                url: 'http://localhost:8070/ContractService/app/openapi/contract/',
                responseType: 'arraybuffer'
            };
            $http(req).success(function(data) {
                $scope.contracts = data.contractHeaders;
            });
        };
    });
