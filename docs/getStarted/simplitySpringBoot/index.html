Simplity with Spring BOOT
Implementation of Simplity with Spring Boot.
Steps to create Simplity application with Spring Boot
1. Get simplity kernel
For Maven users-
  • Copy code
<dependency>
<groupId>org.simplity<groupId>
<artifactIdId>simplity</artifactId>
<version>1.2</version>
</dependency>
Plain Java-
  • Copy code
<dependency>
<groupId>org.simplity</groupId>
<artifactIdId>simplity</artifactId>
<version>1.2</version>
</dependency>
2. Creating project and Configuration
  • Create a new "Maven Project" in eclipse
  • Create component folder structure under resources as follows:
    • comp
      • dt
      • msg
      • openapi
      • rec
      • service
        • tp
      • sql

  • Configure application.xml under resources/comp as follows.
    • Copy code
    <?xml version="1.0" encoding="UTF-8"?>
    <application
            xmlns="http://www.simplity.org/schema"
            applicationId="ContractService"
            autoLoginUserId="100"
            logSqls="true">
    </application>
      
    • This is the root file of simplity which helps to configure various features of Simplity. Here "applicationId" is a mandatory field which represents the application name. applicationId should be a unique name for the application.
    • NOTE: To get more help, click ctrl + tab from the eclipse editor.
  • Create Open API under resources/comp/openapi as follows.
    • Copy code

  • Create Simplity record under resources/comp/rec as follows.
    • Copy code
    <?xml version="1.0" encoding="UTF-8" standalone="no"?>
    <record name="ContractHeader"
            recordType="storage"
            tableName="CONTRACTHEADER"
            keyToBeGenerated="false"
            okToSelectAll="true"
            xmlns="http://www.simplity.org/schema">
      <fields>
        <field columnName="BP_CONTRACT_NUM" dataType="_text" fieldType="primaryKey" name="bpContractNum"         sqlTypeName="VARCHAR"/>
        <field columnName="DESCRIPTION" dataType="_text" fieldType="data" name="description" sqlTypeName="VARCHAR"/>
        <field columnName="ASSET_NAME" dataType="_text" fieldType="data" name="assetName" sqlTypeName="VARCHAR"/>
      </fields>
    </record>

    • Simplity records are kept under the comp/rec folder. This is the main part of data model. Data structures that are used as input to, or output from a service are modeled as records. Following are the fields of simplity record:
    • name (mandatory field) - name of the record or entity
    • moduleName (optional field) - sub-folder name inside rec folder. In this example(rec/login/user.xml) record name is user.xml and module name login.
    • recordType (mandatory field) - STORAGE (represents a row of a table being stored in an rdbms), VIEW (a view involving one or more storage records), STRUCTURE (record structure used by the application, but is not linked to any stored data).
    • tableName (optional field) - name of the db table, if recordType is STORAGE or VIEW.
    • keyToBeGenerated (optional field) - true/false (if any table column value to be generated automatically)
    • okToSelectAll (optional field) - true/false(set true to select all rows from the table)
  • Create Simplity service under resources/comp/service/tp as follows.
    • Copy code
    <?xml version="1.0" encoding="UTF-8" standalone="no"?>
    <service xmlns="http://www.simplity.org/schema"
            name="contractFind"
            dbAccessType="readOnly">
        <actions>
            <filter filterRecordName="filteredContractHeaders" outputSheetName="contractHeaders">
            </filter>
            <log names="contractHeaders"/>
        </actions>
        <outputData>
            <outputRecords>
                <outputRecord sheetName="contractHeaders" />
            </outputRecords>
        </outputData>
    </service>

3. Write Spring Boot main program
  • Write Java program and integrate spring boot,
    • Copy code

    import java.io.File;
    import org.simplity.kernel.Application;
    import org.simplity.rest.Operations;
    import org.slf4j.Logger;
    import org.slf4j.LoggerFactory;
    import org.springframework.boot.SpringApplication;
    import org.springframework.boot.autoconfigure.EnableAutoConfiguration;
    import org.springframework.boot.autoconfigure.web.DispatcherServletAutoConfiguration;
    import org.springframework.boot.web.servlet.ServletRegistrationBean;
    import org.springframework.context.annotation.Bean;
    import org.springframework.context.annotation.ComponentScan;
    import org.springframework.context.annotation.Configuration;
    import org.springframework.web.servlet.DispatcherServlet;

    @Configuration
    @ComponentScan
    @EnableAutoConfiguration
    public class App {
        final static Logger logger = LoggerFactory.getLogger(App.class);

        @Bean
        public DispatcherServlet dispatcherServlet() {
            return new DispatcherServlet();
        }

        @Bean
        public ServletRegistrationBean dispatchServletRegistration() {
            ServletRegistrationBean registration = new ServletRegistrationBean(dispatcherServlet(), "/*");
            registration.setName(DispatcherServletAutoConfiguration.DEFAULT_DISPATCHER_SERVLET_REGISTRATION_BEAN_NAME);
            return registration;
        }

        @Bean
        public ServletRegistrationBean servletRegistrationBean() {
            ServletRegistrationBean registration = new ServletRegistrationBean(new org.simplity.rest.Serve(), "/app/*");
            registration.setName("RestSimplity");
            return registration;
        }

        public static void main(String[] args) throws Exception {
            File jarPath = new File(App.class.getProtectionDomain().getCodeSource().getLocation().getPath());
            String folder = jarPath.getParent() + File.separator + "classes/comp" + File.separator;
            try {
                Application.bootStrap(folder);
                Operations.loadAll(folder + "/openapi/");
            } catch (Exception e) {
                logger.error("error while bootstrapping with compFolder=" + folder, e);
                return;
            }
            SpringApplication.run(App.class, args);
        }
    }

3. Run Spring Boot application
  • In eclipse, right-click on the project --> Run As --> Java Application (Select App.java as Main program)
  • Enter the below URL from the web browser,
    • Copy code
    "http://localhost:8070/ContractService/app/openapi/contract/"

    • Note :
      • ContractService - application name.
      • /app/openapi - basePath.
      • /contract - path to the particular service.
